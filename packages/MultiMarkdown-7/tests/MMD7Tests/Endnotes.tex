\part{Reference Style}
\label{referencestyle}

foo \footnote{This is a footnote.}.

foo ~\citep{cite}.

foo \gls{*glossary*}.

\part{Inline Style}
\label{inlinestyle}

foo \footnote{This is \emph{another} footnote.}.

foo ~\citep{thisisanothercitation.}.

foo \newglossaryentry{*gls*}{name=\emph{gls}, description={This is \emph{another} glossary term.}}\gls{*gls*}

\part{Extended Reference Style}
\label{extendedreferencestyle}

foo \footnote{This is a footnote.

\begin{quote}
With a \emph{tail} block.
\end{quote}}.

foo ~\citep{cite3}.

foo \gls{glossary3}.
